def initialize(edges,store = :list)
  case store
  when :matrix
    @g = GraphAsMatrix.new(edges)
  when :tri_matrix
    @g = GraphAsTriMatrix.new(edges)
  else
    @g = GraphAsList.new(edges)
  end
end

# File ds/graphs/graph.rb, line 46
def bfs s
  colors = {}
  parents =  {}
  res = []
  d = {}
  q = Queue.new

  @g.each_vertex do |v|
    colors[v] = :white
    parents[v] = nil
    d[v] = Infinity
  end

  colors[s] = :white
  parents[s] = nil
  d[s] = 0

  q.enqueue s

  while !q.empty?
    u = q.dequeue 
    @g.neighbors(u).each do |v|
      if colors[v] === :white
        colors[v] = :grey
        d[v] = d[u] + 1
        parents[v] = u
        q.enqueue v
      end
    end
    colors[u] = :black
    res << u
  end
  res
end

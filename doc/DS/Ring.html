def self.from_array(arr)
  list = Ring.new(arr.shift)
  tail = list.head
  arr.each { |e| tail = tail.append(e) }
  tail.next = list.head
  list
end

def eliminate_by(k)
  elem = self.head
  prev = elem

  k = k-1

  while prev != elem.next 

    k.times  do
      prev = elem
      elem = elem.next
    end

    prev.next = elem.next
    elem = prev.next
  end

  return prev.data

end
